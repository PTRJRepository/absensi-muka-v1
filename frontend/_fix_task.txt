# Fix Absensi Muka Frontend — API Mismatch Fix

## PROBLEM
Frontend API client expects `{success: true, data: [...]}` wrapper, but the Absensi_Muka backend sends raw arrays `[{...}]` for most endpoints (NOT wrapped).

## FILES TO FIX

### 1. `D:/Gawean Rebinmas/Absensi_Muka/frontend/src/lib/api.ts`
Replace the entire file content with:

```typescript
const API_BASE = import.meta.env.VITE_API_BASE_URL ?? 'http://localhost:8004';

export function getToken() { return localStorage.getItem('token'); }
export function setToken(token: string) { localStorage.setItem('token', token); }
export function clearToken() { localStorage.removeItem('token'); }

interface ApiResponse<T> {
  success: boolean;
  data: T;
  message?: string;
  error?: { message?: string };
}

export async function api<T>(path: string, options: RequestInit = {}): Promise<T> {
  const response = await fetch(`${API_BASE}${path}`, {
    ...options,
    headers: {
      'Content-Type': 'application/json',
      ...(getToken() ? { Authorization: `Bearer ${getToken()}` } : {}),
      ...(options.headers ?? {}),
    },
  });

  if (!response.ok) {
    throw new Error(`HTTP ${response.status}: ${response.statusText}`);
  }

  const text = await response.text();
  if (!text) return [] as unknown as T;

  let payload: unknown;
  try {
    payload = JSON.parse(text);
  } catch {
    return [] as unknown as T;
  }

  // Handle wrapped responses: { success: true, data: [...] }
  if (
    typeof payload === 'object' &&
    payload !== null &&
    'success' in payload &&
    'data' in payload
  ) {
    const p = payload as ApiResponse<T>;
    if (!p.success) throw new Error(p.error?.message ?? 'API error');
    return p.data;
  }

  // Handle raw array / object responses (no wrapper)
  return payload as T;
}
```
